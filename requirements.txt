# run.py - Application entry point
import os
import logging
from logging.handlers import RotatingFileHandler
from flask import Flask
from config import config
from pm_internship_engine import PMInternshipRecommendationEngine

def create_app(config_name=None):
    """Application factory pattern"""
    if config_name is None:
        config_name = os.environ.get('FLASK_ENV', 'default')
    
    app = Flask(__name__)
    app.config.from_object(config[config_name])
    
    # Setup logging
    if not app.debug:
        if not os.path.exists('logs'):
            os.mkdir('logs')
        
        file_handler = RotatingFileHandler(
            app.config['LOG_FILE'], 
            maxBytes=10240000, 
            backupCount=10
        )
        file_handler.setFormatter(logging.Formatter(
            '%(asctime)s %(levelname)s: %(message)s [in %(pathname)s:%(lineno)d]'
        ))
        file_handler.setLevel(logging.INFO)
        app.logger.addHandler(file_handler)
        
        app.logger.setLevel(logging.INFO)
        app.logger.info('PM Internship Recommender startup')
    
    # Initialize recommendation engine
    recommendation_engine = PMInternshipRecommendationEngine()
    app.recommendation_engine = recommendation_engine
    
    return app

if __name__ == '__main__':
    app = create_app()
    app.run(host='0.0.0.0', port=5000)

# data_preprocessing.py - Data preprocessing utilities
import pandas as pd
import numpy as np
import re
from datetime import datetime
import logging

logger = logging.getLogger(__name__)

class DataPreprocessor:
    """Utility class for preprocessing internship and candidate data"""
    
    def __init__(self):
        self.state_mappings = {
            'andhra pradesh': 'ap', 'arunachal pradesh': 'ar', 'assam': 'as',
            'bihar': 'br', 'chhattisgarh': 'ct', 'goa': 'ga', 'gujarat': 'gj',
            'haryana': 'hr', 'himachal pradesh': 'hp', 'jharkhand': 'jh',
            'karnataka': 'ka', 'kerala': 'kl', 'madhya pradesh': 'mp',
            'maharashtra': 'mh', 'manipur': 'mn', 'meghalaya': 'ml',
            'mizoram': 'mz', 'nagaland': 'nl', 'odisha': 'or', 'punjab': 'pb',
            'rajasthan': 'rj', 'sikkim': 'sk', 'tamil nadu': 'tn',
            'telangana': 'tg', 'tripura': 'tr', 'uttar pradesh': 'up',
            'uttarakhand': 'uk', 'west bengal': 'wb'
        }
    
    def clean_text(self, text):
        """Clean and normalize text data"""
        if pd.isna(text) or text is None:
            return ''
        
        text = str(text).lower().strip()
        # Remove special characters but keep spaces and alphanumeric
        text = re.sub(r'[^a-zA-Z0-9\s]', ' ', text)
        # Remove extra whitespaces
        text = re.sub(r'\s+', ' ', text)
        
        return text
    
    def standardize_location(self, location):
        """Standardize location names"""
        if pd.isna(location) or location is None:
            return ''
        
        location = self.clean_text(location)
        
        # Check if it's a major city
        major_cities = {
            'mumbai': 'mumbai', 'delhi': 'delhi', 'bangalore': 'bangalore',
            'bengaluru': 'bangalore', 'hyderabad': 'hyderabad',
            'chennai': 'chennai', 'kolkata': 'kolkata', 'pune': 'pune',
            'ahmedabad': 'ahmedabad', 'jaipur': 'jaipur', 'surat': 'surat'
        }
        
        for city, standard in major_cities.items():
            if city in location:
                return standard
        
        # Check for states
        for state, code in self.state_mappings.items():
            if state in location:
                return state
        
        return location
    
    def extract_skills(self, skills_text):
        """Extract and standardize skills"""
        if pd.isna(skills_text) or skills_text is None:
            return []
        
        skills_text = self.clean_text(skills_text)
        
        # Common skill mappings
        skill_mappings = {
            'python': 'python', 'java': 'java', 'javascript': 'javascript',
            'html': 'html', 'css': 'css', 'react': 'react', 'nodejs': 'nodejs',
            'sql': 'sql', 'mysql': 'mysql', 'mongodb': 'mongodb',
            'excel': 'excel', 'powerpoint': 'powerpoint', 'word': 'word',
            'communication': 'communication', 'teamwork': 'teamwork',
            'leadership': 'leadership', 'problem solving': 'problem_solving'
        }
        
        found_skills = []
        for skill, standard in skill_mappings.items():
            if skill in skills_text:
                found_skills.append(standard)
        
        return found_skills
    
    def categorize_education(self, education):
        """Categorize education level"""
        if pd.isna(education) or education is None:
            return 'other'
        
        education = self.clean_text(education)
        
        if any(term in education for term in ['10th', 'ssc', 'class 10']):
            return 'class_10'
        elif any(term in education for term in ['12th', 'hsc', 'class 12']):
            return 'class_12'
        elif any(term in education for term in ['diploma', 'polytechnic']):
            return 'diploma'
        elif any(term in education for term in ['iti']):
            return 'iti'
        elif any(term in education for term in ['ba', 'bsc', 'bcom', 'bba', 'bca']):
            return 'bachelors'
        elif any(term in education for term in ['btech', 'be', 'engineering']):
            return 'engineering'
        elif any(term in education for term in ['mba', 'mca', 'msc', 'ma']):
            return 'masters'
        else:
            return 'other'
    
    def validate_internship_data(self, df):
        """Validate internship dataset"""
        required_columns = ['title', 'company', 'sector', 'location', 'description']
        missing_columns = [col for col in required_columns if col not in df.columns]
        
        if missing_columns:
            raise ValueError(f"Missing required columns: {missing_columns}")
        
        # Check for empty values
        for col in required_columns:
            empty_count = df[col].isna().sum()
            if empty_count > 0:
                logger.warning(f"Column '{col}' has {empty_count} empty values")
        
        return True
    
    def preprocess_internships_dataset(self, df):
        """Preprocess the entire internships dataset"""
        logger.info(f"Preprocessing {len(df)} internships")
        
        # Validate data
        self.validate_internship_data(df)
        
        # Clean text columns
        text_columns = ['title', 'description', 'skills_required']
        for col in text_columns:
            if col in df.columns:
                df[f'{col}_clean'] = df[col].apply(self.clean_text)
        
        # Standardize locations
        if 'location' in df.columns:
            df['location_clean'] = df['location'].apply(self.standardize_location)
        
        # Extract skills
        if 'skills_required' in df.columns:
            df['skills_list'] = df['skills_required'].apply(self.extract_skills)
        
        # Add duration in months (assuming format like "12 months", "1 year")
        if 'duration' in df.columns:
            df['duration_months'] = df['duration'].apply(self._extract_duration_months)
        
        logger.info("Preprocessing completed successfully")
        return df
    
    def _extract_duration_months(self, duration_text):
        """Extract duration in months from text"""
        if pd.isna(duration_text) or duration_text is None:
            return 12  # Default to 12 months
        
        duration_text = str(duration_text).lower()
        
        # Look for patterns like "12 months", "1 year", "6 month"
        if 'month' in duration_text:
            months = re.findall(r'(\d+)', duration_text)
            if months:
                return int(months[0])
        elif 'year' in duration_text:
            years = re.findall(r'(\d+)', duration_text)
            if years:
                return int(years[0]) * 12
        
        return 12  # Default
